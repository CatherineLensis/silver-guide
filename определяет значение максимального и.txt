    /*определяет значение максимального и минимального элементов, а также среднее арифметическое значение элементов
    для одномерного массива произвольного числового типа, используя шаблон функции:
 */
#include <iostream>

template <typename T>
T calculateMinMaxAverage(const T* arr, int size, T& minValue, T& maxValue) {
    if (size == 0) {
        // Если массив пуст, возвращаем 0
        return 0;
    }

    // Инициализируем минимальное и максимальное значение первым элементом массива
    minValue = arr[0];
    maxValue = arr[0];
    T sum = arr[0];

    // Проходимся по остальным элементам массива
    for (int i = 1; i < size; i++) {
        // Обновляем минимальное и максимальное значение при необходимости
        if (arr[i] < minValue) {
            minValue = arr[i];
        } else if (arr[i] > maxValue) {
            maxValue = arr[i];
        }

        // Суммируем элементы
        sum += arr[i];
    }

    // Возвращаем среднее арифметическое значение
    return sum / size;
}

int main() {
    const int size = 5;
    int arr[size] = {7, 2, 9, 4, 6};
    int minValue, maxValue;
    double average;

    average = calculateMinMaxAverage(arr, size, minValue, maxValue);

    std::cout << "Минимальное значение: " << minValue << std::endl;
    std::cout << "Максимальное значение: " << maxValue << std::endl;
    std::cout << "Среднее арифметическое: " << average << std::endl;

    return 0;
}
